合併排序法 (Merge Sort)

合併排序是一種高效的分治排序算法。

算法原理

合併排序使用分治法，將陣列分成兩半，遞歸排序，然後合併兩個已排序的子陣列。

JavaScript 實現

function mergeSort(arr) { if (arr.length <= 1) return arr; const mid = Math.floor(arr.length / 2); const left = mergeSort(arr.slice(0, mid)); const right = mergeSort(arr.slice(mid)); return merge(left, right); } function merge(left, right) { let result = []; let i = 0, j = 0; while (i < left.length && j < right.length) { if (left[i] < right[j]) { result.push(left[i++]); } else { result.push(right[j++]); } } return result.concat(left.slice(i)).concat(right.slice(j)); } // 範例使用 let array = [64, 34, 25, 12, 22, 11, 90]; console.log("原始陣列:", array); let sorted = mergeSort(array); console.log("排序後:", sorted);
